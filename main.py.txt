import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from statsmodels.tsa.seasonal import seasonal_decompose
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_absolute_error, mean_squared_error

np.random.seed(42)
dates = pd.date_range(start='2023-01-01', periods=200, freq='D')
trend = np.linspace(0, 20, 200)
seasonal = 10 * np.sin(2 * np.pi * dates.dayofyear / 7)
noise = np.random.normal(0, 2, 200)
target = 50 + trend + seasonal + noise

df = pd.DataFrame({'date': dates, 'target': target}).set_index('date')

def perform_eda(df):
    print("--- Performing EDA ---")
    # Decomposition
    result = seasonal_decompose(df['target'], model='additive', period=7)
    result.plot()
    plt.tight_layout()
    plt.show()

    # Rolling Statistics (Anomalies/Patterns)
    plt.figure(figsize=(10, 4))
    plt.plot(df['target'], label='Original', alpha=0.5)
    plt.plot(df['target'].rolling(7).mean(), label='7-Day Rolling Mean', color='red')
    plt.title("Trend & Rolling Mean")
    plt.legend()
    plt.show()

perform_eda(df)

def create_features(df):
    df = df.copy()
    # Time-based features
    df['day_of_week'] = df.index.dayofweek
    df['month'] = df.index.month
    # Lag features (Memory)
    df['lag_1'] = df['target'].shift(1)
    df['lag_7'] = df['target'].shift(7)
    # Rolling features (Local trend)
    df['rolling_mean_3'] = df['target'].shift(1).rolling(window=3).mean()
    return df.dropna()

data = create_features(df)

split_idx = int(len(data) * 0.8)
train = data.iloc[:split_idx]
test = data.iloc[split_idx:]

X_train = train.drop('target', axis=1)
y_train = train['target']
X_test = test.drop('target', axis=1)
y_test = test['target']

baseline_preds = test['lag_1']

model = RandomForestRegressor(n_estimators=100, random_state=42)
model.fit(X_train, y_train)
model_preds = model.predict(X_test)

def evaluate(y_true, y_pred, label):
    mae = mean_absolute_error(y_true, y_pred)
    rmse = np.sqrt(mean_squared_error(y_true, y_pred))
    print(f"[{label}] MAE: {mae:.2f} | RMSE: {rmse:.2f}")
    return mae, rmse

print("\n--- Model Comparison ---")
b_mae, b_rmse = evaluate(y_test, baseline_preds, "Baseline (Naive)")
m_mae, m_rmse = evaluate(y_test, model_preds, "Random Forest")

plt.figure(figsize=(12, 5))
plt.plot(y_test.index, y_test, label='Actual', color='black', lw=2)
plt.plot(y_test.index, model_preds, label='Model Prediction', linestyle='--', color='blue')
plt.title("Final Model vs Actual (Test Set)")
plt.legend()
plt.show()
